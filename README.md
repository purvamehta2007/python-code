print("LETS GET STARTED WITH OUR CALCULATOR" )
number=int(input("please enter the number : "))
number1=int(input("please enter the number : "))
sign= str(input("please enter the number (+,-,*,/): "))
if sign=="+":
    result=number+number1
    print(result)
if sign=="-":
    result=number-number1
    print(result)
if sign=="*":
    result=number*number1
    print(result)
if sign=="/":
    result=number/number1
    print(result)
